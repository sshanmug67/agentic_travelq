# Create Security Group.
# 1. Create security group
aws ec2 create-security-group --group-name travelq-sg --description "TravelQ Backend Security Group" --output table

# 2. # Get your public IP address first (This is what the Internet Service Provider assigned to the Router at home network)
(Invoke-WebRequest -Uri https://checkip.amazonaws.com).Content.Trim()

# Replace YOUR_IP with what you got above
aws ec2 authorize-security-group-ingress --group-name travelq-sg --protocol tcp --port 22 --cidr 96.231.64.246/32

# HTTP
aws ec2 authorize-security-group-ingress --group-name travelq-sg --protocol tcp --port 80 --cidr 0.0.0.0/0

# HTTPS
aws ec2 authorize-security-group-ingress --group-name travelq-sg --protocol tcp --port 443 --cidr 0.0.0.0/0


# to Reset the local laptop IP Address in security Group:
# If SSH stops working, update the rule with your new IP:
# 1. Get your new public IP
(Invoke-WebRequest -Uri https://checkip.amazonaws.com -UseBasicParsing).Content.Trim()

# 2. Remove old rule
aws ec2 revoke-security-group-ingress --group-name travelq-sg --protocol tcp --port 22 --cidr 96.231.64.246/32

# 3. Add new rule
aws ec2 authorize-security-group-ingress --group-name travelq-sg --protocol tcp --port 22 --cidr NEW_IP/32
